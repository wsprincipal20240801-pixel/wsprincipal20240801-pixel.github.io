const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetLinks = ss.getSheetByName("Links"); function doGet(e) { const action = e.parameter.action; const id = e.parameter.id; let result = {}; try { if (action === "getLink") { result = getLinkById(id); } else if (action === "getAll") { result = getAllLinks(); } else { result = { status: "error", message: "Unknown action" }; } } catch (err) { result = { status: "error", message: err.toString() }; } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { let data; try { data = JSON.parse(e.postData.contents); } catch (err) { data = e.parameter; } const action = data.action; let result = {}; try { if (action === "create") { result = createNewLink(data); } else if (action === "addClick") { result = incrementClick(data.id); } else if (action === "delete") { result = deleteLink(data.id); } else if (action === "updateTitle") { // ★★★ 新增：更新標題的功能 ★★★ result = updateLinkTitle(data.id, data.newTitle); } else { result = { status: "error", message: "Unknown action" }; } } catch (err) { result = { status: "error", message: err.toString() }; } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } // --- 功能函式區 --- function getLinkById(id) { const data = sheetLinks.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == id) { return { status: "success", url: data[i][1], title: data[i][2] }; } } return { status: "error", message: "Link not found" }; } function incrementClick(id) { const data = sheetLinks.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == id) { let currentClicks = data[i][4] || 0; sheetLinks.getRange(i + 1, 5).setValue(currentClicks + 1); return { status: "success", clicks: currentClicks + 1 }; } } return { status: "error", message: "ID not found" }; } function getAllLinks() { const data = sheetLinks.getDataRange().getValues(); const rows = []; for (let i = 1; i < data.length; i++) { if (data[i][0]) { rows.push({ shortId: data[i][0], originalUrl: data[i][1], title: data[i][2], createdDate: data[i][3], clicks: data[i][4], note: data[i][5] }); } } return { status: "success", data: rows.reverse() }; } function createNewLink(payload) { const url = payload.url; let shortId = payload.shortId; let title = payload.title; const note = payload.note || ""; const data = sheetLinks.getDataRange().getValues(); const existingIds = data.map(r => r[0]); if (!shortId) { shortId = generateRandomId(existingIds); } else { if (existingIds.includes(shortId)) { return { status: "error", message: "ID already exists" }; } } if (!title) title = fetchPageTitle(url); sheetLinks.appendRow([shortId, url, title, new Date(), 0, note]); return { status: "success", shortId: shortId, title: title }; } function generateRandomId(existingIds) { const chars = "abcdefghijkmnpqrstuvwxyz23456789"; let result = ""; do { result = ""; for (let i = 0; i < 5; i++) { result += chars.charAt(Math.floor(Math.random() * chars.length)); } } while (existingIds.includes(result)); return result; } function fetchPageTitle(url) { try { const response = UrlFetchApp.fetch(url, {muteHttpExceptions: true}); const html = response.getContentText(); const match = html.match(/(.*?)<\/title>/i); if (match && match[1]) return match[1].trim(); } catch (e) { return ""; } return ""; } function deleteLink(id) { const data = sheetLinks.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] == id) { sheetLinks.deleteRow(i + 1); return { status: "success" }; } } return { status: "error", message: "ID not found" }; } // ★★★ 新增：更新標題實作 ★★★ function updateLinkTitle(id, newTitle) { const data = sheetLinks.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == id) { // 標題在第 3 欄 (C欄)，也就是 index 2，但在 getRange 要用 3 sheetLinks.getRange(i + 1, 3).setValue(newTitle); return { status: "success" }; } } return { status: "error", message: "ID not found" }; }
